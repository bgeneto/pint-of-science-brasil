import csv
from app.db import db_manager
from app.models import Auditoria

def exportar_logs_csv(arquivo_saida="logs_export.csv"):

                    log.acao,
                    log.entidade,
                    log.entidade_id,
                    log.detalhes
                ])

        print(f"‚úÖ {len(logs)} logs exportados para {arquivo_saida}")
